(function($) {
$(function() {
  $(".rating").rating({
    focus: function(value, link) {
      $("#test-tip").html(link.title || link.value);
    },
    blur: function(value, link) {
      $("#test-tip").html('&nbsp;');
    }
  });
});
})(jQuery);
</script>
